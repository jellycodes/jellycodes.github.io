def stack_algorithm():
    num = int(input())
    order = []
    result = []
    for i in range(num):
        stack = (input())
        if 'push' in stack:
            order.append(stack.split()[-1])
        elif 'pop' in stack:
            if order:
                result.append(order.pop())
            else:
                result.append(-1)
        elif 'size' in stack:
            result.append(len(order))
        elif 'empty' in stack:
            if not order:
                result.append(1)
            else:
                result.append(0)
        elif 'top' in stack:
            if order:
                result.append(order[-1])
            else:
                result.append(-1)
    for i in range(len(result)):
        print(result[i])

stack_algorithm()
