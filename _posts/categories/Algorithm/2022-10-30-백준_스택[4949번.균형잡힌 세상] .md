result = []
while True:
    input_ = input()
    stack = []
    if input_ == '.':
        break
    for i in input_:
        if i == '(' or i == '[':
            stack.append(i)
        elif i == ')':
            if stack and stack[-1] == '(':
                stack.pop()
            else:
                stack.append(')')
                break
        elif i == ']':
            if stack and stack[-1] == '[':
                stack.pop()
            else:
                stack.append(']')
                break
    if not stack and input_[-1] == '.':
        result.append('yes')
    else:
        result.append('no')

for res in result:
    print(res)
