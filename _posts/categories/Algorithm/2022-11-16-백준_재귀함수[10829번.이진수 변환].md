from collections import deque
result = deque()
def dec_to_bin(N):
    if N % 2 == 1 and N != 1:
        N = N // 2
        result.appendleft('1')
        print(N)
        dec_to_bin(N)
    elif N % 2 == 0:
        N = N // 2
        result.appendleft('0')
        print(N)
        dec_to_bin(N)
    else:
        result.appendleft('1')
        print(''.join(result))
        
N = int(input())
dec_to_bin(N)
